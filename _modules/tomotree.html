Image Reconstruction of Internal Defects in Wood Based on Segmented

number of segmentes such that the lengths of the segments are closer than

def _find_ellipse_vertices(major_start: numpy.ndarray, major_end: numpy.ndarray,
                           eccentricity: float):
    """
    Given end points on major axis and eccentricity, find the vertices coordiantes.
    Parameters
    ----------
    major_start : numpy.ndarray
        One endpoint on major axis
    major_end : numpy.ndarray
        another end point on major axis
    eccentricity : float
        Eccentricity, the number between 0 and 1

    Returns
    -------
    A : numpy.ndaray
        
    """
    center = 0.5 * ( major_end + major_start )
    half_axis_vector = major_end - center
    half_minor_axis = np.array(
        [half_axis_vector[1],-half_axis_vector[0]]) * np.sqrt( 1 - eccentricity ** 2 ) 
    A = np.array([major_start, major_end, 
                  center - half_minor_axis, center + half_minor_axis])
    return A
    
def _find_segments_through(s: object, 
                           allsegments: numpy.ndarray,
                           points_on_segments: list = None,
                           eccentricity: float = 0.98, 
                           include_weight: bool = False, 
                           check_only_rectagle = False, 
                           delta_factor = 0.5) -> list:
    """
    For segment s find all segments with nonzero intersection with ellipse around segment s.
    The segments are in the array allsegments.
    The test is performed for points from the array points_on_segments.

    If include_weight is True, each segment is in the answer so many times, how many points are in the
    intersection with the ellipse. If not, each segment is in the answer only once.
    
    To save the computation time, we first remove from consideration the segments
    which are outside the rectangle surrrounding the ellipse. The size of the rectangle is 
    defined by the value of `delta_factor` variable. If `check_only_rectagle`
    is True, only this rectangle is considered and the ellipse is skipped.
    """
    
    M = _find_ellipse_vertices(s.start_coordinates, s.end_coordinates, eccentricity)
    delta = delta_factor * s.length
    idx = ( (points_on_segments[...,0] > M[:,0].min()-delta) & 
            (points_on_segments[...,0] < M[:,0].max()+delta) &
            (points_on_segments[...,1] > M[:,1].min()-delta) &
            (points_on_segments[...,1] < M[:,1].max()+delta) )
    idx = idx.any(axis=-1)
    allsegments = allsegments[idx]           
    if check_only_rectagle:
        return allsegments
    if points_on_segments is not None:
        points_on_segments = points_on_segments[idx]
    else:
        starts = np.array(
            [seg.start_coordinates for seg in allsegments]).reshape(-1, 1, 2)
        ends = np.array(
            [seg.end_coordinates for seg in allsegments]).reshape(-1, 1, 2)
        points_on_segments = starts + \
                             np.linspace(0, 1, 10).reshape(1, -1, 1) * (ends - starts)
    # idx = s.test_inside_ellipse(points_on_segments, eccentricity=eccentricity)
    idx = is_point_inside_ellipse(points_on_segments, s.start_coordinates, s.end_coordinates,  eccentricity=eccentricity)
    if include_weight:
        # VARIANT B:
        # include each segments so many times, how many points from the
        # segment are in the intersection
        important_segments = np.array(
            sum([[i] * w for i, w in zip(allsegments, idx.sum(axis=1))], start=[]))
    else:
        # VARIANT A:
        # include only segments with nonzero intersection, each segment once
        idx = idx.any(axis=-1)
        important_segments = allsegments[idx]
    return important_segments

def speed2color(speed: float, max: float = None, min: float = None, speed_min: float = None, speed_max: float = None):
    """Convert speed to color.
    Return black, if the speed is outside minimal and maximal value.
    Othervise return a color according to the colormap scheme.

    * Parameters ``min`` and ``max`` set the minimal and maximal value.
    * If the values are not given, values ``speed_min`` and ``speed_max`` are used.
    * If none of these values are given, set ``min=0`` and ``max=1``.

    Parameters
    ----------
    speed : float
        Speed to be converted to color
    max: float
        Maximal value, the speed correspoding to the fastest color
    min: float
        Minimal value, the speed corresponding to the color with smallest velocity.
    speed_min: float
        The values `speed_min` and `speed_max` are used if `min` and `max` are not set.
    speed_max: float
    """
    if max is None:
        max = speed_max
    if min is None:
        min = speed_min
    if max is None:
        max = 1
    if min is None:
        min = 0
    if speed is None:
        return None
    if speed > max or speed < min:
        return "#000000"
    return cmap((speed - min) / (max - min))

def angle_between(v1: numpy.ndarray, v2: numpy.ndarray) -> float:
    """Calculate and return the signed angle in radians between two vectors `v1` and `v2`.

    Parameters
    ----------
    v1 : numpy.ndarray
        The first input vector. It should be a 2D vector.
    v2 : numpy.ndarray
        The second input vector. It should be a 2D vector.

    Returns
    -------
    float
        The signed angle in radians between ``v1`` and ``v2``. The angle is
        measured counterclockwise from ``v1`` to ``v2``.


    The function first calculates the angle of ``v1`` from the positive x-axis (``angle1``).
    Then rotates ``v2`` by ``-angle1`` to align `v1` with the positive x-axis.
    Finally, it calculates the angle of the rotated ``v2`` from the positive 
    x-axis, which is the signed angle between ``v1`` and ``v2``.
    """
    angle1 = -np.arctan2(v1[1], v1[0])
    v2_rotated = np.array(
        [[np.cos(angle1), -np.sin(angle1)], [np.sin(angle1), np.cos(angle1)]]) @ v2
    angle = np.arctan2(v2_rotated[1], v2_rotated[0])
    return angle

def get_boundary_points(N: int = 5,
                        dx: float = 1, dy: float = 1, posunx: float = 0, posuny: float = 0) -> numpy.ndarray:
    """The function `get_boundary_points` is designed to generate points on 
    the boundary of a rectangle. This can be useful for testing if a cell is 
    within an elliptical domain of a given ray.

    Parameters
    ----------
    N
        The number of points to generate on each side of the rectangle.
        The default value is 5.
    dx
        The length of the rectangle in the x-direction. The default
        value is 1.
    dy
        The length of the rectangle in the y-direction. The default
        value is 1.
    posunx
        The x-coordinate of the bottom-left corner of the rectangle. The
        default value is 0.
    posuny
        The y-coordinate of the bottom-left corner of the rectangle. The
        default value is 0.


    The function returns a 2D numpy array with shape (2, 4*N). The first 
    row of the array contains the x-coordinates of the points, and the 
    second row contains the y-coordinates. The points are ordered 
    counterclockwise, starting from the bottom-left corner of the rectangle.
    """
    t = np.linspace(0, 1, N, endpoint=False)
    data = np.zeros((2, 4 * N))
    # First row, x coordinates
    data[0, :N] = t * dx
    data[0, N:2 * N] = dx
    data[0, 2 * N:3 * N] = dx * (1 - t)
    # Second row, y coordinates
    data[1, N:2 * N] = t * dy
    data[1, 2 * N:3 * N] = dy
    data[1, 3 * N:4 * N] = dy * (1 - t)
    return data.T + np.array([posunx, posuny])

def is_point_inside_ellipse(point: numpy.ndarray, start: float, end: float, eccentricity: float = 0.9,
                            distance_from_vertex: bool = False) -> numpy.ndarray:
    """Test if a point is inside the ellipse. The ellipse is defined by the
    start and end point of the main axis and eccentricity. If ``distance_from_vertex``
    is true, returns also the length of the distance between the points and
    the start vertex measured in the direction of the major axis. (Useful for
    spatial interpolation by the segmented ellipse.)

    Parameters
    ----------
    point
        Point or an array of points to be tested. The last axis of the array 
        is supposed to have length 2.
    start
        One of the ends of the major axis.
    end
        The second ends of te major axis.
    eccentricity
        Ellipse eccentricity. The default is 0.9.
    distance_from_vertex
        If true, return also distance from vertex, i.e. return pair
        [boolean, distance from minor]
        
    Returns
    -------
    bool
        Returns bool or array of bools if ``distance_from_vertex`` is ``False``.
    list
        Returns a pair of bool and float if ``distance_from_vertex`` is ``True``.
    """
    center = (start + end) / 2
    a = np.linalg.norm(center - end)
    # See https://en.wikipedia.org/wiki/Semi-major_and_semi-minor_axes
    # for length of minor axis from eccentricity and major axis
    b = a * np.sqrt(1 - eccentricity ** 2)
    # Distance from the minor axis is the length of the projection
    # of the vector from center to the point
    # to the direction of major axis. And vice versa.
    # The vector of minor axis is perpendicular
    # to the vector of major axis.
    vector = point - center
    e =  (end-center)/np.linalg.norm(end - center)
    perp_e = perp(e)
    # The following two lines make the code slow. Consider to change the 
    # approach to solve the task.
    distance_from_minor = np.dot(vector, e)
    distance_from_major = np.dot(vector, perp_e)
    ans = (distance_from_minor) ** 2 * b ** 2 + (distance_from_major) ** 2 * a ** 2 <= (a*b)**2
    if distance_from_vertex:
        ans = [ans, distance_from_minor + a]
    return ans



def _c_value_EBSI(i: int, j: int, N: int) -> float:
    """Return c_ij value base on the paper Stress Wave Tomography of Wood Internal Defects
    using Ellipse-Based Spatial Interpolation and Velocity Compensation, Du et al.,
    Bioresources (2015),  BioResources 10(3), 3948-3962, formula (4)

    Parameters
    ----------
    i : int
        node number
    j : int
        node number
    N : int
        number of nodes

    Returns
    -------
    unknown
        Formula (4) from Du et al. (2015). “Stress wave tomography,”
        BioResources 10(3), 3948-3962.
    """
    if np.abs(i - j) == N / 2:
        return 0.1
    if np.abs(j - i) < N / 2:
        return 1 - np.abs(j - i) / (N / 2)
    return 1 - (N - np.abs(j - i)) / (N / 2)


def _e_value_EBSI(*args):
    """Return eccentricity from according to formula (4) from Du et al.
    (2015). “Stress wave tomography,” BioResources 10(3), 3948-3962.

    Parameters
    ----------
    *args
        parametres node number, node number, number of nodes are passed
        to `_c_value_EBSI`
    """
    return np.sqrt(1 - _c_value_EBSI(*args) ** 2)

def roll_table(data: numpy.ndarray):
    """Converts table with n columns as soures and (n-1) rows as receivers
    to n times n table.
    Input is table in which the first row is the TOF to the first neighbor
    etc. 
    Output is table where each row contains TOF of the same node.
    """
    d = pd.DataFrame(
        index=range(1, data.shape[0] + 2),
        columns=range(1, data.shape[1] + 1)
    )
    d.index.name = "Receiver node"
    d.columns.name = "Source node"
    d.iloc[1:, :] = data
    for j in d.columns:
        d.loc[:, j] = np.roll(d.loc[:, j], j - 1)
    return d

class Cell:
    """Class for cells in grid. Contains the list of rays and their velocities (or velocities on the
    corresponding ray segment) and the velocity

    Attributes
    ----------
    rays : list
        A list of rays passing through the cell.
    velocity : float
        The velocity of the cell.
    velocities_list : list
        List of velocities of rays or segments which go through
        the cell. This list is used inside loops. In the first step in the

        of the cell. 
    x : float
        The x-coordinate of the bottom left corner of the cell.
    y : float
        The y-coordinate of the bottom left corner of the cell.
    dx : float
        The width of the cell.
    dy : float
        The height of the cell.

        List of points on boundary. Useful when testing if a cell is
        inside ellipse.
    segments : set

    id : int or None
        The id of the cell. Initiated when creating the grid.

        self.y = y
        self.dx = dx
        self.dy = dy
        self.rays = rays

        return f"Cell {self.id} with velocity {self.velocity}.\nNumber of rays trough: {len(self.rays)}."

class Grid:
    """Class representing a grid in a tomogram.

    Cells are stored in a dataframe. This allows to get the cell by using
    a coordinates of a point inside using :code:`loc` method.

    Attributes
    ----------
    x : float
        Array containing the x-coordinates of the grid.
    y : float
        Array containing the y-coordinates of the grid.
    cells : pandas.DataFrame
        DataFrame representing the grid cells, initialized with Cell
        objects.
    all_cells : list
        List of cells in the diagram.
    """

    def __init__(self, xlim=(0, 10), nx=11, ylim=(0, 10), ny=11):
        """Initializes a Grid object. Also fills the `Grid.all_cells()` list
        and sets the `Cell.id` for cells in this list.
        """
        self.x = np.linspace(*xlim, nx)
        self.y = np.linspace(*ylim, ny)
        self.dx = self.x[1] - self.x[0]
        self.dy = self.y[1] - self.y[0]
        data = [[Cell(x=j, y=i, dx=self.dx, dy=self.dy)
                 for i in self.y] for j in self.x]
        self.cells = pd.DataFrame(data=data,
                                  index=self.x, columns=self.y)
        self.cells.iloc[-1, :] = np.nan
        self.cells.iloc[:, -1] = np.nan
        self.cells.index.name = "x"
        self.cells.columns.name = "y"
        self.all_cells = [
            i for i in self.cells.values.reshape(-1) if not isinstance(i, float)]
        for i, cell in enumerate(self.all_cells):
            cell.id = i

    def __str__(self):
        """Returns a string representation of the Grid object.

        Returns:
            str: A string representing the dimensions of the grid.
        """
        return f"Grid from {len(self.x) - 1}x{len(self.y) - 1} cells."

    def get_cell(self, x: float, y: float) -> Cell:
        """Retrieves a cell from the grid based on its coordinates.

        Args:
            x (float):
                The x-coordinate of the cell.
            y (float):
                The y-coordinate of the cell.

        Returns:
            Cell:
                The cell object located at the specified coordinates.
        """
        # Get the last x node which does not exceed the x-coordinate given
        x_coord = self.x[self.x <= x][-1]
        # Get the last y node which does not exceed the y-coordinate given
        y_coord = self.y[self.y <= y][-1]
        return self.cells.at[x_coord, y_coord]



    def _save_rays_info_to_cells(self, rays):
        """Save list of rays passing through to each cell of the grid."""
        for ray in rays:
            # get list of cells on each ray
            param = np.linspace(0, 1, 200).reshape(-1, 1)
            points = ray.start.coordinates + param * \
                     (-ray.start.coordinates + ray.end.coordinates)
            cells = {self.get_cell(*point) for point in points}
            # add information about the ray to the cell atribute
            for cell in cells:
                if cell is np.nan:
                    continue
                if ray not in cell.rays:
                    cell.rays = cell.rays + [ray]

    def set_cell_velocity_from_ray_means(self):
        """Update cell velocities. Takes list of rays through the cell.
        The velocities of the rays from this list are used and the new
        cell velocity is calculated as mean of the velocities of the rays.
        """
        def find_mean_velocity(cell):
            if cell is np.nan:
                # The cell is not in the tree area
                # Return undefined cell
                return cell
            if len(cell.rays) == 0:
                # There is no ray through this cell
                # Return cell with undefined velocity
                return np.nan
            # There are rays through the cell.
            # Return the average of the speeds.
            data = [i.speed for i in cell.rays if i.speed is not None]
            if len(data) == 0:
                cell.velocity = np.nan
            else:
                cell.velocity = np.mean(data)
            return cell

        self.cells.map(find_mean_velocity)

class Ray:
    """Ray in the acoustic tomography. 

    When initialized, the ray is added to the ``rays_from`` and ``rays_to`` 
    attributes of the node at the start and at the end.

    Attributes
    ----------
    start : Node
        The starting node of the ray.
    end : Node
        The ending node of the ray.
    speed : float, optional
        The speed of the ray. If not provided, it will be calculated
        based on the distance between start and end nodes and the given
        time.
    time : float, optional
        The time taken for the ray to travel from start to end. Only
        used if speed is not provided.
    segments : set
        The set of segments for the segmented method.
    angle_from_radial : float
        The angle between the ray and the radial direction.
    length : float
        The length of the ray.
    """

    def __init__(self, start: Node, end: Node, speed = None, time = None):
        """Initialize a Ray object.

        Parameters:
        - start (Node): The starting node of the ray.
        - end (Node): The ending node of the ray.
        - speed (float, optional): The speed of the ray. If not provided, it will be calculated based on the distance between start and end nodes and the given time.
        - time (float, optional): The time taken for the ray to travel from start to end. Only used if speed is not provided.
 
        Returns:
        - None
        """
        # Save start and end node and speed
        self.start = start
        self.end = end
        self.length = np.sqrt(
            np.sum((start.coordinates - end.coordinates) ** 2))
        if speed is not None:
            self.speed = speed
            self.time = self.length / self.speed
        elif time is not None:
            self.speed = self.length / time
            self.time = time
        else:
            self.speed = None
            self.time = None
        # Save the ray to the nodes info
        self.start.rays_from.add(self)
        self.end.rays_to.add(self)
        self.start.get_ray_to[end] = self
        self.segments = set([Segment(self, speed=self.speed, start=0, end=1)])
        self.angle_from_radial = angle_between(
            start.coordinates, start.coordinates - end.coordinates)

    def __str__(self):
        return f"Ray from node {self.start.number} to node {self.end.number}. Speed {np.round(self.speed, 0)}, {len(self.segments)} segments."

    """Ray segment. Properties are parent ray, the speed on the segment and the start
    and the end position of the segment on the ray. The start of the ray
    corresponds to the position 0, the end corresponds to position 1.

    Attributes
    ----------
    ray : Ray
        The parent ray of the segment.
    speed : float
        The speed on the segment.
    start : float
        The start position of the segment on the ray. Float from 0 to 1.
    end : float
        The end position of the segment on the ray. Float from 0 to 1.
    start_coordinates : array-like
        The start coordinates of the segment.
    end_coordinates : array-like
        The end coordinates of the segment.
    length : float
        length of the segment

        self.ray = ray
        self.speed = speed
        self.start = start
        self.end = end
        self.start_coordinates = ray.start.coordinates + \
                             start * (ray.end.coordinates - ray.start.coordinates)
        self.end_coordinates = ray.start.coordinates + end * \

    def split(self, n=2):
        """Split the segment to two or more segments. Removes the segment 
        from the ray.segments set and adds new segments to this set.

        Return the set from the new segments. This allows to adjust the speed on
        these new segments.
        """
        if n == 1:
            return set([self])
        t = np.linspace(self.start, self.end, n + 1)
        new_segments = [Segment(ray=self.ray, start=t[i], speed=self.speed, end=t[i + 1])
                        for i in range(n)]
        self.ray.segments.remove(self)
        self.ray.segments = self.ray.segments.union(set(new_segments))
        return new_segments



    @property
    def length(self) -> float:
        """Return length of the segment"""
        return np.linalg.norm(self.start_coordinates - self.end_coordinates)

class Tomo:
    """Class for constructing and displaying a tomogram or generally for processing 
    the output from a tomograph.

    The arguments are a list of vertices and a table with speeds.

    By default, there are 12 points evenly distributed on a circle.

    Parameters
    ----------
    nodes : list, optional
        List of pairs ``[[x1, y1], [x2, y2], ...]`` representing
        coordinates of nodes. If ``None``, nodes are generated regularly on
        a circle. The default is ``None``.
    speeds : numpy.ndarray, optional
        The table with speeds. The times are calculated from speeds and lengths. 
        The data from each source is in a column. The data arriving to sensors
        are in the rows. The table is square table with NaN in main diagonal.
        The default is None.
    times : numpy.ndarray, optional
        The table with times if speeds are not given. The speeds are calculated
        from the times and lengths. The default is None. The format (source/receiver)
        is the same as for speeds.
    nx : int
        The number of cells along horizontal axis.
    ny : int
        The number of cells along vertical axis.
    symmetric : bool
        If True, the ray speed on each ray is an average of speeds in
        both directions. This allows to speed up the processing when the
        method is symmetric. The default is False.
    eps: float
        The gap between the lefmost (rightmost, top bottom) node and the 
        grid boundary. Should be nonzero for SIRT.
        
    Attributes
    ----------
    nodes : list
        The list of nodes in the tomogram.
    rays : list
        The list of rays in the tomogram.
    speeds :
        The speed table. 
    times :
        The table of times. 
    get_node_by_number : dict
        Dictionary which allows to access the node by a number of the node.
    n :
        The number of vertices
    grid :
        The grid for the tomogram
    polygon :
        The boundary of the tomogram. Polygon with vertices in nodes.
    get_node_by_number :
        Dict which allows to access nodes in tomogram by their numbers.
    """

    def __init__(self, nodes=None, speeds=None, times=None, nx=21, ny=21, symmetric=False, eps=0.01):
        """Initialize Tomo class."""
        if times is not None and isinstance(times, pd.DataFrame):
            times = times.values
        if nodes is None and speeds is not None:
            nodes_number = speeds.shape[0]
        elif nodes is None and times is not None:
            nodes_number = times.shape[0]
        else:
            nodes_number = 12
        if nodes is None:
            nodes = [np.array([np.sin(2 * np.pi / nodes_number * i), np.cos(2 * np.pi / nodes_number * i)])
                     for i in range(nodes_number)]
        self.nodes = [Node(node, i + 1) for i, node in enumerate(nodes)]
        self.get_node_by_number = {i.number: i for i in self.nodes}
        if speeds is not None:  # tomogram defined by speeds
            if symmetric:
                self.rays = [Ray(self.nodes[i], self.nodes[j], speed=np.mean([speeds[i, j], speeds[j, i]]))
                             for i in range(len(nodes)) for j in range(len(nodes)) if i < j]
            else:
                self.rays = [Ray(self.nodes[i], self.nodes[j], speed=speeds[j, i])
                             for i in range(len(nodes)) for j in range(len(nodes)) if i != j]
        elif times is not None:  # tomogram defined by times
            if symmetric:
                self.rays = [Ray(self.nodes[i], self.nodes[j], time=np.mean([times[i, j], times[j, i]]))
                             for i in range(len(nodes)) for j in range(len(nodes)) if i < j]
            else:
                self.rays = [Ray(self.nodes[i], self.nodes[j], time=times[j, i])
                             for i in range(len(nodes)) for j in range(len(nodes)) if i != j]
        else:  # epmty tomogram
            if symmetric:
                self.rays = [Ray(self.nodes[i], self.nodes[j], time=None)
                             for i in range(len(nodes)) for j in range(len(nodes)) if i < j]
            else:
                self.rays = [Ray(self.nodes[i], self.nodes[j], time=None)
                             for i in range(len(nodes)) for j in range(len(nodes)) if i != j]
        self.__symmetric = symmetric
        self.n = len(nodes)
        nodes_x = [i[0] for i in nodes]
        nodes_y = [i[1] for i in nodes]
        self.grid = Grid(xlim=[min(nodes_x) - eps, max(nodes_x) + eps], nx=nx,
                         ylim=[min(nodes_y) - eps, max(nodes_y) + eps], ny=ny)

        self.grid._save_rays_info_to_cells(self.rays)
        self.grid.set_cell_velocity_from_ray_means()

        self.polygon = shapely.geometry.polygon.Polygon(
            [i.coordinates for i in self.nodes])

    def __str__(self):
        ans = f"Tomo with {self.n} nodes and {len(self.rays)} rays divided into {len(self.allsegments)} segments." + \
              f" The grid contains {len(self.grid.x) - 1}x{len(self.grid.y) - 1} cells."
        return ans

    @property
    def speeds(self):
        nodenumbers = [i.number for i in self.nodes]
        nodenumbers.sort()
        df = pd.DataFrame(index=nodenumbers, columns=nodenumbers)
        for i in self.nodes:

                try:
                    df.at[i.number, j.number] = j.get_ray_to[i].speed

                    df.at[i.number, j.number] = np.nan
        df.index.name = "Receiver node"
        df.columns.name = "Source node"
        return df

    @property

        df = pd.DataFrame(index=nodenumbers, columns=nodenumbers)
        for i in self.nodes:

                try:
                    df.at[i.number, j.number] = j.get_ray_to[i].time

                    df.at[i.number, j.number] = np.nan
        df.index.name = "Receiver node"
        df.columns.name = "Source node"
        return df

        return [seg for ray in self.rays for seg in ray.segments]

    def fix_velocity_EBSI(self, function=lambda x, y: x / (1 - 0.2 * np.abs(y ** 2))):
        """Fix the velocities as described in EBSI in the paper 
        `Du et al. (2015). “Stress wave tomography,” BioResources 10(3), 3948-3962. <https://bioresources.cnr.ncsu.edu/resources/stress-wave-tomography-of-wood-internal-defects-using-ellipse-based-spatial-interpolation-and-velocity-compensation/>`_

        * The function used to fix the velocities can be customized, 
          if necessary. The default is :code:`lambda x,y: x/(1-0.2*np.abs(y**2)`.
        * The function does not allow to be called twice on the same 
          tomogram. Prints a friendly notice if the user attempts to call 
          is for the second time. The variable 
          :code:`self.velocities_EBSI_fixed` is set to :code:`True`.
        """
        try:
            if self.velocities_EBSI_fixed:
                logging.debug(
                    "Friendly reminder: Tangential velocities for ESBI already fixed, skipping this step.")
        except AttributeError:
            for ray in self.rays:
                ray.speed = function(ray.speed, ray.angle_from_radial)
                for seg in ray.segments:
                    seg.speed = ray.speed
            self.velocities_EBSI_fixed = True

    def do_EBSI_method(self, fix_velocity_EBSI_function=lambda x, y: x / (1 - 0.2 * np.abs(y ** 2))):
        """Do the EBSI method on tomogram. The tomogram object is updated. 

        - Clear the :code:`velocities_list` of cells. This removes any data 
          from previous methods, if any.
        - Fix the velocity of tangent rays, if required and if not done
          before. This fixes the fact that the speed depends on the angle
          between the ray and the radial direction.
        - For each ray get the list of cells influenced by the ray. For each cell
          influenced by the ray add the ray speed to the  `velocities_list`. 
          Use points on cell boundaries and ellipse defined by the ray and 
          eccentricity to check, if the cell is influenced by the ray.
        - Update cell velocities as mean of the values from :code:`velocities_list`
          from the previous step. Use weighted mean, the weights are lengths 
          of segments.
        - The cells outside the polygon are ignored.

        Parameters
        ----------
        fix_velocity_EBSI_function
            Allows to set the function f(speed,angle) which describes
            how to fix the rays having declination `angle` from radial
            direction. The default :code:`lambda x,y:
            x/(1-0.2*np.abs(y**2))` setting is from the paper `Du et al.
            (2015). “Stress wave tomography,” BioResources 10(3),
            3948-3962.
            <https://bioresources.cnr.ncsu.edu/resources/stress-wave-
            tomography-of-wood-internal-defects-using-ellipse-based-
            spatial-interpolation-and-velocity-compensation/>`_
        Returns
        -------
        dict
            A dictionary where keys are rays and values are lists of cells
            influenced by the ray.
        """
        start_time = time.time()
        self.grid._clear_velocities_list()
        self.fix_velocity_EBSI(fix_velocity_EBSI_function)
        # For each ray find the cells affected by the ray, i.e. get the
        # cells inside the ellipse around the ray. Then store the ray
        # speed to the velocities list in the cell.
        c = np.array(self.grid.all_cells)
        cells_bp = np.array(list(map(lambda x: x.boundary_points, c)))
        logging.info(f"EBSI method: cells_bp has shape {cells_bp.shape}.")
        N = len(self.nodes)
        ans = {}
        for ray in self.rays:
            test_points_affected = is_point_inside_ellipse(cells_bp,
                ray.start.coordinates, ray.end.coordinates,
                _e_value_EBSI(ray.start.number, ray.end.number, N))
            test_cells_affected = test_points_affected.any(axis=1)
            c_subset_affected = c[test_cells_affected]
            logging.info(f"EBSI method: Ray {ray.start.number}-{ray.end.number} affects {len(c_subset_affected)} cells.")
            for i in c_subset_affected:
                i.velocities_list += [{'velocity': ray.speed, 'weight': 1 / ray.length}]
            ans[ray] = c_subset_affected
        self._update_cell_velocity(use_weights=True)
        logging.debug(f"EBSI method finished in {time.time() - start_time}s.")
        return ans



    def _do_ERSM_ray_segmentation(self, n: float = 3):
        """Do the ray segmentation for ERSM method.

        Each ray is divided into segments. The shortest ray is divided
        into ``n`` segments. The other rays are divided into similar lengths.
        Parameters
        ----------
        n
            The shortest ray is divided into ``n`` segments. The other rays are 
            divided into similar lengths.
        """
        # Do the segmentation.
        # Find length for segmentation
        lengths = [ray.length for ray in self.rays]
        dl = min(lengths) / n
        # Do the ray segmentation
        for seg in self.allsegments:
            length = np.linalg.norm(
                seg.start_coordinates - seg.end_coordinates)
            n = length // dl
            seg.split(int(n))

    def do_ERSM_method(self,
                       fix_velocity_EBSI_function=lambda x, y: x /
                                                               (1 - 0.2 * np.abs(y ** 2)),
                       e_segmentation: float = 0.98,
                       e_interpolation: float = 0.99,
                       n = 2,
                       mean = np.nanmean):
        """Do the ERSM method on tomogram.

        * Clear the ``velocities_list`` of cells. This removes any 
          relicts from previous methods.
        * Fix the velocity of tangent rays if required. This fixes the fact that 
          the speed in radial direction is maximal 
        * Do the ray segmentation
        * For each segment on ray find the cells affected by the segment, i.e.
          get the cells inside the ellipse around the ray such that the
          orthogonal projection to major axis is in the segment. Then store
          the segment speed to the ``velocities_list`` in the cell.
        * For each ray get the list of cells influenced by the ray and then
          update the ``velocities_list``
        * Update cell velocities as mean of the values from ``velocities_list``
          from the previous step
        * The cells outside the polygon are ignored.
        
        We set the eccentricity values for both ray segmentation and spatial
        interpolation. We use eccentricity rather than the quotient of minor
        and major axes, the value ``c`` value in Du (2018). The conversion is
        by the formula ``e=sqrt(1-c**2)``.

        Parameters
        ----------
        fix_velocity_EBSI_function
            Allows to set the function f(speed,angle) which describes
            how to fix the rays having declination `angle` from radial
            direction. The default :code:`lambda x,y:
            x/(1-0.2*np.abs(y**2))` setting is from the paper `Du et al.
            (2015). “Stress wave tomography,” BioResources 10(3),
            3948-3962.
            <https://bioresources.cnr.ncsu.edu/resources/stress-wave-
            tomography-of-wood-internal-defects-using-ellipse-based-
            spatial-interpolation-and-velocity-compensation/>`_
        e_segmentation
            Eccentricity for ray segmentation algorithm. This eccentricity
            is used when setting the velocities of the ray segments. Du suggests
            c value for this step between 0.65 and 0.8 which is eccentricity
            between 0.76 and 0.60.
        e_interpolation
            Eccentricity for spatial interpolation by segmented ellipse.
            Du suggests c value between 0.85 and 0.95 which is e between
            0.53 and 0.31.
        n
            The shortest ray is divided into ``n`` segments. The other rays are 
            divided into similar lengths.
        """
        logging.debug("ERSM started")
        start_time = time.time()
        N = len(self.nodes)
        self.grid._clear_velocities_list()  # Clear the velocities list
        # Fix the values on nonradial directions
        self.fix_velocity_EBSI(fix_velocity_EBSI_function)
        # Do the segmentation
        self._do_ERSM_ray_segmentation(n=n)
        
        # Find all segments and generate points on these segments
        allsegments = np.array(self.allsegments) # np.aray is used to speed up the code
        starts = np.array(
            [s.start_coordinates for s in allsegments]).reshape(-1, 1, 2)
        ends = np.array(
            [s.end_coordinates for s in allsegments]).reshape(-1, 1, 2)
        points_on_segments = starts + \
            np.linspace(0, 1, 10).reshape(1, -1, 1) * (ends - starts)
        logging.debug(f"    ERSM segmentation finished in {time.time() - start_time:.3f}s.")
        # Fix the speeds on segments
        # for _ in allsegments:
        #     important_segments = _find_segments_through(
        #         _, allsegments, points_on_segments, eccentricity=e_segmentation)
        #     _.speed = np.mean([i.ray.speed for i in important_segments])
        q = map(lambda _segment: mean([i.ray.speed for i in 
                  _find_segments_through(
                      _segment, 
                      allsegments, 
                      points_on_segments, 
                      eccentricity=e_segmentation)]), 
            allsegments)
        for i,j in zip (q,allsegments):
            j.speed = i
        logging.debug(f"    ERSM segment speed determinantion finished in {time.time() - start_time:.3f}s.")
        # For each segment on ray find the cells affected by the segment, i.e. get the
        # cells inside the ellipse around the ray such that the orthogonal projection to
        # major axis is in the segment. Then store the segment speed to the velocities
        # list in the cell.
        c = np.array(self.grid.all_cells)
        cells_bp = np.array(list(map(lambda x: x.boundary_points, c)))
        for ray in self.rays:
            idx_r, d_r = is_point_inside_ellipse(
                cells_bp, 
                start=ray.start.coordinates, end=ray.end.coordinates,
                eccentricity=e_interpolation, 
                distance_from_vertex=True)
            segments = list(ray.segments)
            # segments.sort(key = lambda x:x.start)
            for i, s in enumerate(segments):
                idx2 = (d_r > s.ray.length * (s.start)
                        ) & (d_r < s.ray.length * (s.end))
                c_subset_affected = c[(idx_r & idx2).any(axis=1)]
                for i in c_subset_affected:
                    i.velocities_list += [s.speed]
        logging.debug(f"    SISE finished in {time.time() - start_time:.3f}s.")
        self._update_cell_velocity(mean=mean)
        logging.debug(f"ERSM method finished in {time.time() - start_time:.3f}s.")



    def _update_cell_velocity(self, use_weights: bool = False, mean=np.nanmean):
        """For each cell test if the cell is in the polygon defined
        by the nodes. If it is, set the velocity of the cell as
        a mean of the speeds in the velocity_list of the cell.
        
        If `use_weights` is True, the `Cell.velocity` is assumed to be dict 
        with `velocity` and `weight` keys.
        """
        for cell in self.grid.all_cells:
            point = shapely.geometry.Point(
                cell.x + 0.5 * cell.dx, cell.y + 0.5 * cell.dy)
            if len(cell.velocities_list) > 0 and self.polygon.contains(point):
                if use_weights:
                    velocities = np.array([i['velocity'] for i in cell.velocities_list])
                    weights = np.array([i['weight'] for i in cell.velocities_list])
                    ans = sum(velocities * weights) / sum(weights)
                else:
                    ans = mean(cell.velocities_list)
                cell.velocity = ans
            else:
                cell.velocity = np.nan

    def update_segments_from_cells(self):
        """Split rays to segments. The boundary between segments is at the 
        cell boundaries. The speed is determined by the cell velocity. If the 
        cell velocity is undefined (may appear near the boundary), the average
        speed on the ray is used."""
        for r in self.rays:
            xs, ys = r.start.coordinates
            xe, ye = r.end.coordinates
            xi = self.grid.x
            yi = self.grid.y
            ti = np.append((xi - xs) / (xe - xs),(yi - ys) / (ye - ys))
            idx = (ti >= 0.0) & (ti <= 1.0)
            ti = ti[idx]
            ti = np.unique(np.append(ti, [0, 1]))
            ti.sort()
            r.segments = [Segment(r, r.speed, ti[i], ti[i + 1]) for i in range(len(ti) - 1)]
        for s in self.allsegments:
            center = ( s.start_coordinates + s.end_coordinates ) / 2
            cell_at_center = self.grid.get_cell(*center)
            if not pd.isna(cell_at_center.velocity):
                s.speed = cell_at_center.velocity

    def do_SIRT_method(self, speed_max: float = 1300, speed_min: float = 300, fix_reaction_time: bool = False):
        """SIRT method as described in SIRT- and CG-Type Methods for the Iterative Solution
        of Sparse Linear Least-Squares Problems by Sluis and Vorst 
        LINEAR ALGEBRA AND ITS APPLICATIONS 130:257-302 (1990)."""

        def _fix_solution(x, speed_min, speed_max, fix_reaction_time=False):
            """Fix the vector of slownesses during SIRT iterations. The velocities 
            should be between maximal and minimal value. Optionaly exclude
            the last element from this adjusting.
            """
            # Save the last value if necessary
            if fix_reaction_time:
                save = x[-1]
            # Fix the values outside bounds
            x[x < 1 / speed_max] = 1 / speed_max
            x[x > 1 / speed_min] = 1 / speed_min
            # Restore the last value if necessary
            if fix_reaction_time:
                x[-1] = save
            return x

        def _solve_SIRT(A, b, x, n=100):
            """Solve the system Ax=b using SIRT method with n iterations."""
            sum_col = 1 / A.sum(axis=0)
            C = np.diag(sum_col)
            sum_row = 1 / A.sum(axis=1)
            R = np.diag(sum_row)
            CATR = C @ (A.T) @ R
            CATRb = CATR @ b
            CATRA = CATR @ A
            for i in range(100):
                corr = CATRb - CATRA @ x
                x = x + corr
                x = _fix_solution(x, speed_min, speed_max)
            return x

        def _find_segments(self):
            """Split rays to segments. The boundary between segments is at the 
            cell boundaries."""
            for r in self.rays:
                xs, ys = r.start.coordinates
                xe, ye = r.end.coordinates
                xi = self.grid.x
                yi = self.grid.y
                ti = np.append((xi - xs) / (xe - xs),(yi - ys) / (ye - ys))
                idx = (ti >= 0.0) & (ti <= 1.0)
                ti = ti[idx]
                ti = np.unique(np.append(ti, [0, 1]))
                ti.sort()
                r.segments = [Segment(r, r.speed, ti[i], ti[i + 1]) for i in range(len(ti) - 1)]

        def _setup_matrices(self, fix_reaction_time: bool):
            """Setup matrices for SITR method. Returns matrices A,b,x from the system
            Ax=b, where b are times of flight, A is matrix of segment lengths and x
            uknown slownesses.
            
            If `fix_reaction_time` is True, adds column of 1 at the end of the matrix A and 
            adds one unknown.
            """
            d = np.zeros((len(self.rays), len(self.grid.all_cells)))
            times = np.array([r.length / r.speed for r in self.rays])
            for i, r in enumerate(self.rays):
                for s in r.segments:
                    d[i, s._storage.id] = s.length
            # remove zero columns from the consideration    
            mask = (d == 0).all(0)
            A = d[:, ~mask]
            active_cells = np.array(self.grid.all_cells)[~mask]
            self.active_cells = active_cells
            x_len = len(active_cells)
            if fix_reaction_time:
                x_len += 1
                A = np.hstack((A, np.ones((A.shape[0], 1))))
            x = np.zeros(x_len)
            b = times
            return A, b, x, active_cells

        start = time.time()
        _find_segments(self)  # Split rays to segments.
        for s in self.allsegments:
            center = (s.end_coordinates + s.start_coordinates) / 2
            cell = self.grid.get_cell(*center)
            s._storage = cell
            cell.segments += [s]

        A, b, x, active_cells = _setup_matrices(self, fix_reaction_time=fix_reaction_time)
        x = _solve_SIRT(A, b, x)

        if fix_reaction_time:
            A = A[:, :-1]
            reaction_time = x[-1]
            sol = x[:-1]
        else:
            reaction_time = None
            sol = x
        self._matrices = {"A": A, "b": b}

        for s, c in zip(sol, active_cells):
            seg = [i for i in self.allsegments if c == i._storage]
            c.velocity = 1 / s
            for sg in seg:
                sg.speed = 1 / s

        report = (
            f"SIRT finished in {1000 * (time.time() - start):.0f} ms."
            f"\nSpeed interval: min={min([i.speed for i in self.allsegments])}"
            f" for {sum(np.array([i.speed for i in self.allsegments]) == speed_min)} segments"
            f", max={max([i.speed for i in self.allsegments])} "
            f" for {sum(np.array([i.speed for i in self.allsegments]) == speed_max)} segments"
        )
        if fix_reaction_time:
            report += f"\nReaction time is {reaction_time * 1000} ms."

        return report

    def limit_cell_plot_velocities(self, fraction: float = 3) -> dict:
        """" Find limits suitable for tomogram plot.

        Parameters
        ----------
        self :
            Tomogram to be potted
        fraction : float
            Fraction of the total number of cells, which should have maximal and minimal colors. The value 0.1 ensures that
            10 percent of the cells will have minimal color and 10 percent maximal color.

        Returns
        -------
            Return dictionary with keys `speed_min` and `speed max`. The author is expected to pass this dictionary to
            the plot method of the tomograf.
        """
        cells_vel = np.array(
            [i.velocity for i in self.grid.all_cells if not isinstance(i, float) and not np.isnan(i.velocity)])
        cells_vel.sort()
        cells_n = len(cells_vel)
        limits = {'speed_min': cells_vel[int(cells_n / fraction)], 'speed_max': cells_vel[-int(cells_n / fraction)]}
        return limits

    def plot(self, speed_min = None, speed_max = None, figsize=(6, 6),
             grid: bool = True, rays: bool = True, ax = None, shift_ray: float = 0.01,
             title = None, shading='flat', use_contourf: bool = False, filter_rays = lambda x: True, lw=2, **kwds):
        """Draw the rays with color corresponding to the speed on each ray.
        It is possible to set the interval for the speeds. If not, it is
        determined automatically from the minimal and maximal value.

        The value shift_ray allows to set the shift betwenn rays in the opposite
        direction. This allows to see if the speeds in opposite directions differ.
        """
        # Adjust minimal and maximal speed if not adjusted by the user.
        if speed_min is None or speed_max is None:
            # speed_min = min([min(r.speeds) for r in self.rays])
            # speed_max = max([max(r.speeds) for r in self.rays])
            speeds = np.array([i.speed for i in self.rays])
            speed_min = np.nanmin(speeds)
            speed_max = np.nanmax(speeds)

        if ax is None:
            fig, ax = plt.subplots(figsize=figsize)
        if grid:
            self.grid.plot(ax=ax, speed_min=speed_min,
                           speed_max=speed_max, shading=shading,
                           use_contourf=use_contourf, **kwds)
        else:
            sc = ax.scatter([0, 0], [0, 0], c=[speed_min, speed_max],
                            s=0, cmap=cmap)
            plt.colorbar(sc)

        # Draw the rays
        if self.__symmetric:
            shift_ray = 0
        if rays:
            for s in self.allsegments:
                if not filter_rays(s.speed):
                    continue
                # draw the ray segment
                color = speed2color(s.speed, max=speed_max, min=speed_min)
                direction = s.end_coordinates - s.start_coordinates
                direction = direction / np.linalg.norm(direction) * shift_ray
                shift = perp(direction)
                ax.plot(*list(zip(s.start_coordinates + shift,
                                  s.end_coordinates + shift)), lw=lw, color=color)

        # Draw node labels
        for n in self.nodes:  # node labels
            ax.text(*(1.1 * n.coordinates), n.number, ha='center', va='center')
            ax.plot(*n.coordinates, "ok")
        ax.set_aspect(1)
        if title is not None:
            ax.set_title(title)
        ax.axis('off')
        plt.tight_layout()
        return ax

    def check_triangle_inequality(self):
        """Check if the direct route is shorter (smaller time) than the route
        via another node.
        """
        dft = pd.DataFrame(index=pd.MultiIndex.from_tuples(
            [[i.number, j.number, via.number]
             for i in self.nodes for j in self.nodes for via in self.nodes
             if i != j and i != via and j != via]),
            columns=['direct route', 'part1', 'part2', 'check'])

        for s, e, j in dft.index:
            s = self.get_node_by_number[s]
            e = self.get_node_by_number[e]
            j = self.get_node_by_number[j]
            r = s.get_ray_to[e]
            r1 = s.get_ray_to[j]
            r2 = j.get_ray_to[e]
            direct_time = r.time
            t1 = r1.time
            t2 = r2.time
            dft.loc[(s.number, e.number, j.number), :] = [direct_time, t1, t2, None]
        dft['check'] = dft['part1'] + dft['part2'] > dft['direct route']

        n = dft.shape[0]
        failed = n - sum(dft.check)
        return f"Triangle inequality failed {failed} out of {n} which is {100 * failed / n:.0f}%."

    def optimize_back(self):
        """
        Rescales using linear transformation the veolocities of segments and cells
        to match travel times as close as possible.
        """
        def find_time(ray,k=1,q=0):
            return sum([s.length / (k * s.speed + q) for s in ray.segments])
        
        def sum_res(t, k=1, q=0):
            return sum([(find_time(i,k,q) - i.time)**2 for i in t.rays])

        def velocity_transformation(t,x):
            return sum_res(t,k=x[0],q=x[1])

        x0 = [1,0]
        opt = minimize(
            lambda x:velocity_transformation(self,x), 
            x0=x0, 
            tol=velocity_transformation(self,x0)/10000
            )
        g = self.grid
        k, q = opt.x
        for c in g.all_cells:
            c.velocity = c.velocity * k + q        
        return{'k': k, 'q':q}

def distances_to_coordiantes(df, num=12):
    output = np.zeros((num,2))
    output[num//2,1] = - df.loc[f"{num//2+1}-1"].iloc[0]
    a = df.loc["7-1"].iloc[0]
    for i in range(2,13):
        if i == 7:
            continue
        b = df.loc[f"{i}-1"].iloc[0]
        c = df.loc[f"{i}-7"].iloc[0]
        cos_gamma = ( a**2 + b**2 - c**2 ) / ( 2 * a * b )
        sin_gamma = np.sqrt( 1 - cos_gamma**2 )
        output[i-1,0] =  - b * sin_gamma 
        output[i-1,1] =  -b * cos_gamma 
        if i > 7 :
            output[i-1,0] =  b * sin_gamma 
    return output
