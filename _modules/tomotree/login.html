def login(top_text_sidebar=None):
    """
    Function to handle user login in a Streamlit app.
    Displays a login form if the user is not authenticated, and allows
    them to log in with a password stored in an environment variable.
    If the user is authenticated, it displays a logout button in the sidebar.
    Args:
        top_text_sidebar (str): Optional text to display at the top of the sidebar.
    Returns:
        None: The function modifies the Streamlit session state and does not
        return a value.

    How to run app:
    1. Set the environment variable `APP_PASSWORD` to the desired password.
    2. Run the Streamlit app.
    3. A oneliner to run the app with a password is:
    ```bash
    APP_PASSWORD=password streamlit run app_tomotree.py
    ```
    """
    ENV_PASSWORD = os.getenv("APP_PASSWORD")
    if ENV_PASSWORD is None:
        return None

    # Save the authentication state in the session state
    if "authenticated" not in st.session_state:
        st.session_state.authenticated = False

    # If the user is not authenticated, show the login form
    if not st.session_state.authenticated:
        c = st.columns(4)
        with c[1]:        
            image_path = os.path.join("_static", "tree.png")
            image = Image.open(image_path)
            st.image(image, caption="TomoTree library is a part of the DYNATREE ERC CZ project", use_container_width=True)
        with c[2]:
            st.markdown("## ðŸ”’ Login form")
            password = st.text_input("Zadej heslo:", type="password")
            if st.button("Login", use_container_width=True, type="primary"):
                if password == ENV_PASSWORD:
                    st.session_state.authenticated = True
                    st.rerun()
                else:
                    st.error("Incorrect password. Please try again.")            
        st.stop()
    else:
        with st.sidebar:
            if top_text_sidebar is not None:
                st.markdown(top_text_sidebar)
            if st.button("Logout.", use_container_width=True, type="primary"):
                st.session_state.authenticated = False
                st.rerun()

    return None

def save_log_message(message, filename="log/login.txt"):
    """
    Function to save a log message to a file.

    The format of the log message is:
        yyyy-mm-dd hh:mm:ss - user - message
    The log file is created if it does not exist, and the message is appended to the end of the file.
    Args:
        message (str): The message to be logged.
        filename (str): The name of the file where the message will be saved.
    Returns:
        None: The function does not return a value.
    """
    if not os.path.exists(os.path.dirname(filename)):
        os.makedirs(os.path.dirname(filename))
    with open(filename, "a") as f:
        log_message = f"{datetime.now().strftime('%Y-%m-%d %H:%M:%S')} - {message}\n"
        f.write(log_message)

def stauth_login(*args, **kwds):
    """
    Function to handle user login using Streamlit Authenticator.
    This function is a placeholder and does not implement any authentication logic.
    Returns:
        None: The function does not return a value.
    """
    ENV_PASSWORD = os.getenv("APP_PASSWORD")
    if ENV_PASSWORD is None:
        return None
    import yaml
    from yaml.loader import SafeLoader

    with open('login.yaml') as file:
        config = yaml.load(file, Loader=SafeLoader)

    # Pre-hashing all plain text passwords once
    # stauth.Hasher.hash_passwords(config['credentials'])

    authenticator = stauth.Authenticate(
        config['credentials'],
        config['cookie']['name'],
        config['cookie']['key'],
        config['cookie']['expiry_days']
    )

    c = st.columns(4)
    with c[2]:        
        authenticator.login()
    if not st.session_state.get('authentication_status'):
        with c[1]:        
            image_path = os.path.join("_static", "tree.png")
            image = Image.open(image_path)
            st.image(image, caption="TomoTree library is a part of the DYNATREE ERC CZ project", use_container_width=True)

    if st.session_state.get('authentication_status'):
        with st.sidebar:
            save_log_message(f"User {st.session_state.get('name')} logged in successfully.")
            st.write(f'Welcome *{st.session_state.get("name")}*')
            authenticator.logout(use_container_width=True)
    elif st.session_state.get('authentication_status') is False:
        save_log_message("Login failed: Username/password is incorrect.")
        with c[2]:
            st.error('Username/password is incorrect')
        with open('config2.yaml', 'w') as file:
            yaml.dump(config, file, default_flow_style=False, allow_unicode=True)            
        st.stop()
    elif st.session_state.get('authentication_status') is None:
        save_log_message("Login failed: Username/password is not provided.")
        with c[2]:
            st.warning('Please enter your username and password')
        st.stop()

    with open('config2.yaml', 'w') as file:
        yaml.dump(config, file, default_flow_style=False, allow_unicode=True)
    return authenticator
