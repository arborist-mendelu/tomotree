Resistograph data processing module.

This module provides a class to handle resistograph data processing, including reading,
preprocessing, and removing global linear trends from the data. It also includes a function
to visualize the resistograph data as colored bars on a tomogram image.

This code is part of the Tomotree project.

class Resistograph:
    """
    A class to handle resistograph data processing, including reading, preprocessing, 
    and removing global linear trends from the data.

    Attributes:
    -----------
    directory : str
        Directory containing resistograph data files.
    nan_lower_limit : int
        Number of initial rows to set as NaN.
    nan_upper_limit : int
        Number of rows after which data is set as NaN.
    smooth : bool
        Whether to apply smoothing to the data.
    set_min_treshold : int
        Minimum threshold for valid data.
    window_length : int
        Window length for Savitzky-Golay filter.
    polyorder : int
        Polynomial order for Savitzky-Golay filter.
    data : pandas.DataFrame
        DataFrame containing processed resistograph data.
    """
    directory: str
    nan_lower_limit: int = 0
    nan_upper_limit: int = 22000
    smooth: bool = False
    set_min_treshold: int = 0
    window_length: int = 201
    polyorder: int = 3
    data: pd.DataFrame = field(default=None, init=False)
    def __post_init__(self):
        self.read_resistograph_data()
        self.preprocess_data()

    def read_resistograph_data(self):
        """
        Read resistograph data from files in the specified directory and process it.

        The method reads all files with the extension `.dpa`, extracts the data section, 
        and calculates the square root of the values. The processed data is stored in a 
        pandas DataFrame.
        """
        files = glob.glob(f"{self.directory}/*dpa")
        files.sort()
        data = {}
        logging.debug(f"Reading {len(files)} files from {self.directory}")
        for i, file in enumerate(files):
            df = pd.read_csv(file, header=None)
            hlavicka = np.argmax(df.iloc[:, 0] == '[DATA]')
            data[i] = df[hlavicka + 1:].reset_index(drop=True).astype(float)

        # for i in data.keys():
        #     data[i] = np.sqrt(data[i])
        df = pd.concat(data, axis=1)
        df.columns = [i[0] + 1 for i in df.columns]
        df.index = df.index / 100
        df.index.name = "depth/mm"
        df.columns.name = "position"
        self.data = df

    def remove_global_linear_trend(self):
        """
        Remove the global linear trend from the resistograph data.

        This method calculates the linear trend across all columns and subtracts it 
        from the data to normalize the values.
        """
        x_all = []
        y_all = []

        for col in self.data.columns:
            y = self.data[col]
            mask = y.notna()
            x_all.extend(np.where(mask)[0])
            y_all.extend(y[mask].values)

        if len(x_all) < 2:
            raise ValueError("Not enough valid values to calculate the trend.")

        coeffs = np.polyfit(x_all, y_all, deg=1)
        trend = np.polyval(coeffs, np.arange(len(self.data)))
        self.data = self.data.subtract(trend, axis=0)

    def preprocess_data(self):
        """
        Preprocess the resistograph data by applying various transformations.

        This method reads the data, sets initial and final rows as NaN, applies 
        smoothing if enabled, and removes values below a minimum threshold.
        """
        if len(self.data) == 0:
            self.read_resistograph_data()
        if self.nan_lower_limit > 0:
            self.data.iloc[:self.nan_lower_limit] = np.nan
        self.data.iloc[self.nan_upper_limit:] = np.nan

        if self.smooth:
            self.data = self.data.apply(lambda col: savgol_filter(col, self.window_length, self.polyorder))

        if self.set_min_treshold > 0:
            lb = self.data.iloc[self.set_min_treshold:, :].min().min()
            mask = self.data < lb
            self.data[mask] = np.nan

def add_resistograph_to_image(data, tomogram, ax=None, cax=None, min=100, max=200, zorder=100, columns=None, linewidth=16, 
                                    meter_length=20, step=300, **kwargs):
    """
    Plot resistograph data as series of colored bars on the picture with tomogram.

    This function visualizes resistograph data by overlaying it as colored bars on a tomogram image. 
    The bars are color-coded based on the resistograph values, and a color bar is added to indicate 
    the value range. Additionally, a scale indicator is drawn on the image.

    Parameters
    ----------
    data : pandas.DataFrame
        A DataFrame containing resistograph data. Each column represents a resistograph measurement 
        for a specific node.
    tomogram : object
        An object representing the tomogram. It must have a `get_node_by_number` method that returns 
        nodes with `coordinates` attributes.
    ax : matplotlib.axes.Axes, optional
        The axes on which to plot the resistograph data. If None, a new figure and axes are created.
    cax : matplotlib.axes.Axes, optional
        The axes on which to plot the color bar. If None, the color bar is added to the main axes.
    min : float, optional
        The minimum value for the color normalization. Default is 100.
    max : float, optional
        The maximum value for the color normalization. Default is 200.
    zorder : int, optional
        The z-order for the plotted elements. Default is 100.
    columns : list, optional
        A list of column indices from the DataFrame to plot. If None, all columns are used.
    linewidth : float, optional
        The width of the colored bars. Default is 16.
    meter_length : float, optional
        The length of the scale indicator in meters. Default is 20.
    step : int, optional
        The step size for down-sampling the resistograph data. Default is 300.
    **kwargs : dict
        Additional keyword arguments for customization.
    Returns
    -------
    ax : matplotlib.axes.Axes
        The axes with the plotted resistograph data.
    Notes
    -----

    - The function assumes that the tomogram nodes are numbered sequentially and that their coordinates 
      are accessible via the `get_node_by_number` method.
    - The scale indicator is drawn at the bottom-left corner of the plot, with labels "5cm" and "10cm".
    - Missing or NaN values in the resistograph data are ignored during plotting.
    """
    df = data.copy()
    if ax is None:
        fig, ax = plt.subplots(figsize=(10, 10))
    if cax is None:
        cax = ax
    
    ax.set_aspect(1)

    cmap = 'gray'
    norm = plt.Normalize(min, max)

    if columns is None:
        columns = df.columns 
    for i in columns:
        j = i + 1 if i < 12 else 1

        center = 0.5 * (tomogram.get_node_by_number[i].coordinates + tomogram.get_node_by_number[j].coordinates)

        values = df[i].values
        u = -center / np.linalg.norm(center)
        x = np.arange(len(values)) / 100000 * u[0] + center[0]
        y = np.arange(len(values)) / 100000 * u[1] + center[1]

        xred = np.arange(np.min([10*100*meter_length]), step=5000) / 100000 * u[0] + center[0]
        yred = np.arange(np.min([10*100*meter_length]), step=5000) / 100000 * u[1] + center[1]

        mask = ~np.isnan(values)
        x, y, values = x[mask], y[mask], np.clip(values[mask], min, max)

        points = np.array([x, y]).T.reshape(-1, 1, 2)[::step]
        segments = np.concatenate([points[:-1], points[1:]], axis=1)

        lc = LineCollection(segments, cmap=cmap, norm=norm, linewidth=linewidth, zorder=zorder, alpha=1.0)
        lc.set_array(values[::step])
        ax.add_collection(lc)

        points = np.array([xred, yred]).T.reshape(-1, 1, 2)
        segments = np.concatenate([points[:-1], points[1:]], axis=1)
        colors = ['red' if i % 2 == 0 else 'black' for i in range(len(xred))]

        lc = LineCollection(segments, colors=colors, linewidth=4, zorder=zorder, alpha=1.0)
        ax.add_collection(lc)        

    xlims = ax.get_xlim()
    ylims = ax.get_ylim()
    segments = np.array([
        [[xlims[0], ylims[0]], [xlims[0]+0.05, ylims[0]]], 
        [[xlims[0]+0.05, ylims[0]], [xlims[0]+0.1, ylims[0]]], 
                ])
    colors = ['red', 'black']
    lc = LineCollection(segments, colors=colors, linewidth=8, zorder=zorder, alpha=1.0)
    ax.add_collection(lc)
    # add text "5cm" and "10cm" to the left of the line
    ax.text(xlims[0] + 0.05, ylims[0] - 0.005, '5cm', fontsize=10, color='black', ha='center', va='top')
    ax.text(xlims[0] + 0.1, ylims[0] - 0.005, '10cm', fontsize=10, color='black', ha='center', va='top')
    
    sm = plt.cm.ScalarMappable(cmap=cmap, norm=norm)
    sm.set_array([])
    plt.colorbar(sm, cax=cax)

    return ax

def get_drilling_start(i, nodes_df):
    """
    Get the starting point and angle of the drilling for a given index.

    Parameters
    ----------
    i : int
        Index of the drilling position.
    nodes_df : pd.DataFrame
        DataFrame containing node information.

    Returns
    -------
    dict
        A dictionary containing the coordinates and angle of the drilling start.
    """
    ncols = nodes_df.shape[0]
    j = i + 1 if i + 1 <= ncols else 1
    ox, oy = (nodes_df.loc[[i, j], ['x', 'y']].mean())
    angle = np.degrees(np.arctan2(-oy, -ox))
    return {'coordinates': np.array([ox, oy]), 'angle': angle}

def add_resistograph_graphs(resistograph_df, nodes_df, ax, yshift=100, yscale=20, color='C0'):
    """
    Add resistograph graphs to the plot.

    Parameters
    ----------
    resistograph_df : pd.DataFrame
        DataFrame containing resistograph data.
    nodes_df : pd.DataFrame
        DataFrame containing node information.
    ax : matplotlib.axes.Axes
        Axes object to add the graphs to.
    yshift : int
        Vertical shift for the graphs (default is 100).
    yscale : int
        Vertical scale for the graphs (default is 20).

    Returns
    -------
    matplotlib.axes.Axes
        The axes with the resistograph graphs added.
    """
    resistograph_df = (
        (resistograph_df.index / 1000)
        .to_series()
        .pipe(lambda idx: resistograph_df.set_index(idx))
    )
    resistograph_df = ((resistograph_df - yshift).clip(lower=0))
    resistograph_df = resistograph_df.div(resistograph_df.max()).div(yscale)
    u = resistograph_df.index.values

    for i in resistograph_df.columns:
        driling_start = get_drilling_start(i, nodes_df)
        T = Affine2D().rotate_deg(driling_start['angle']).translate(*driling_start['coordinates'])
        v = resistograph_df[i].values
        trans = T + ax.transData
        ax.plot(u, v, transform=trans, color='C0')
        ax.fill_between(u, v, 0, where=v >= 0, transform=trans,
                        alpha=0.4, zorder=1000, color=color)
    return ax
