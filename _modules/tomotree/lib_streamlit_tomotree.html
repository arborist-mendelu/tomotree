def get_data_directory(dirname="streamlit_data"):
    """Returns the directories where the data files are stored.

    Returns the list of subdirectories of the dirname directory.
    The list is sorted by the directory names.
    The directories are relative to the current working directory.
    """
    import os
    import glob

    # Check if the directory exists
    if not os.path.exists(dirname):
        raise FileNotFoundError(f"The directory '{dirname}' does not exist.")

    # Get all subdirectories in the data directory
    subdirs = [d for d in glob.glob(os.path.join(dirname, "*")) if os.path.isdir(d)]

    # Sort the subdirectories by their names
    subdirs.sort()

    return subdirs

def get_file(directory, mask=None, extension=None, all=False):
    """Returns the first file (or all files) in the directory that matches the mask and extension.
    """
    import os
    import glob

    if not os.path.exists(directory):
        raise FileNotFoundError(f"The directory '{directory}' does not exist.")

    # Create a search pattern based on the mask and extension
    if mask is None:
        pattern = os.path.join(directory, "*")
    else:
        pattern = os.path.join(directory, f"*{mask}*")

    if extension is not None:
        pattern += f".{extension}"

    # Get all files matching the pattern
    files = glob.glob(pattern)

    if not files:
        return None

    files.sort()  # Sort files by name
    if all:
        return files  # Return all matching files
    return files[0]  # Return the first matching file

def get_dpa_data(files, take_sqrt=False):
    """
    Processes DPA data files and returns a concatenated DataFrame.

    Args:
        files (list): List of file paths to process.
        take_sqrt (bool): Whether to apply square root transformation to the data.

    Returns:
        pd.DataFrame: Processed and concatenated data.
    """
    if files is None:
        return None
    data = {}
    for i, file in enumerate(files):
        df = pd.read_csv(file, header=None, sep=r'[\t,]', engine='python')
        header_index = df.iloc[:, 0].eq('[DATA]').idxmax()
        data[i] = df.iloc[header_index + 1:].reset_index(drop=True).astype(float)

    if take_sqrt:
        # Apply square root transformation to the data.
        data = {key: np.sqrt(value) for key, value in data.items()}

    # Concatenate all data into a single DataFrame.
    df = pd.concat(data, axis=1)
    df.columns = range(1, len(df.columns) + 1)
    df.index = df.index / 100
    df.index.name = "depth/mm"
    df.columns.name = "position"
    return df

def get_data_from_directory(directory, normalize_nodes=True):
    """Returns the data from the directory.

    If normalize_nodes is True, the nodes coordinates are normalized
    by subtracting the mean of the coordinates.
    """

    def load_csv(mask=None, extension="csv", header=None):
        file = get_file(directory, extension=extension, mask=mask)
        return pd.read_csv(file, header=header, sep=r'[\t,]', engine='python') if file else pd.DataFrame()

    ans = {
        'f3d_file': get_file(directory, extension='f3d'),
        'nodes': load_csv(mask="nodes"),
        'times': load_csv(mask="times"),
        'speeds': load_csv(mask="speeds"),
        'resistograph': get_dpa_data(get_file(directory, extension='dpa', all=True)),
        'video': get_file(directory, extension='mp4', all=True)
    }

    directory_contents = os.listdir(directory)
    # sort by extension and name
    directory_contents.sort(key=lambda x: (os.path.splitext(x)[1], x))
    # Check for readme files
    for readme in ['readme.txt', 'readme.md', 'README.txt', 'README.md', 'readme', 'README']:
        if readme in directory_contents:
            with open(os.path.join(directory, readme), 'r') as f:
                ans['readme'] = f.read()
    ans['files'] = directory_contents
    if normalize_nodes and ans['nodes'] is not None:
        ans['nodes'] -= ans['nodes'].mean(axis=0)

    if ans['f3d_file'] is not None and len(ans['nodes']) == 0:
        logging.debug("Getting nodes from f3d.")
        root = analyze_f3d.get_root(ans['f3d_file'])
        df_distance = analyze_f3d.extract_distance_data(root)
        ans['nodes'] = pd.DataFrame(analyze_f3d.df_nodes(df_distance, save=False))
        logging.debug("Nodes extracted from f3d file: {ans['nodes']} ")
    return ans

def uploader():
    """
    Not used anymore, but kept for compatibility.

    Handles uploading the files with nodes coordinates, travel times, and ray speeds.
    """
    with st.sidebar:
        """
        ## Predefined data

        Predefined data from Jičín Linden alley. You can upload you own data.
        """

        jicin_tree = st.radio("Tree data", ["05", "08", "09", "11", "14", "15", "neither, upload csv"], index=1,
                              horizontal=True)
        if len(jicin_tree) == 2:
            return jicin_tree, jicin_tree, None, True

        """
        ## File upload

        Upload the files with nodes coordinates and either 
        the file with the travel times or file with ray speeds. 
        """

        """
        ## Coordinates
        """
        df_nodes_file = st.file_uploader("Upload file with coordinates of nodes.", type=['csv', 'f3d'],
                                         label_visibility='collapsed')
        with st.expander("File format description for nodes coordinates"):
            """
            * File is expected to be a csv file. The separator is detected automatically.  
            * Each row contains the coordinates of a node. 
            * No file header is expected. The number of rows corresponds to the number of nodes.
            """
        """
        ## Travel times
        """
        df_times_file = st.file_uploader("Upload file with travel times.", type=['csv'], label_visibility='collapsed')
        with st.expander("File format description for travel times"):
            """
            * File is expected to be a csv file. The separator is detected automatically.  
            * No file header is expected. The number of columns corresponds to the number
            of nodes, the number of rows is either equal to the number of nodes or less by one. 
            * The numbers in each column are the travel times from
            the same source node to all other nodes. The travel times are in
            seconds. 
            * If the number of rows is the same as the number of columns, 
            the data in the main diagonal are supposed to be NaNs (undefined).
            * If the number of rows is smaller, oit is supposed that the number in the 
            first row is TOF to the neighbor node, the second number is TOF to the
            second neighbor, etc. The last number is the TOF to the last node, a
            neighbor on the opposite side.        
            """
        """
        ## Ray speeds
        """
        df_speeds_file = st.file_uploader("Upload file with speeds.", type=['csv'], label_visibility='collapsed')
        with st.expander("File format description for speeds"):
            """See the description of the travel times file format.
            """
    if df_times_file is None and df_speeds_file is None:
        st.warning("""Please upload the file with travel times or speeds and optionally the file 
        with node positions.""")
        footer()
        st.stop()
    return df_nodes_file, df_speeds_file, df_times_file, False

def reader(df_nodes_file, df_speeds_file, df_times_file, isdemo):
    """Not used anymore, but kept for compatibility.
    """
    if isdemo:
        df_nodes_file = f"demo/nodes-{df_nodes_file}.csv"
        df_nodes = pd.read_csv(df_nodes_file, header=None, sep=None)
    # if df_nodes_file is None:
    #     df_nodes_values = None
    #     df_nodes = None
    # else:
    elif df_nodes_file.name[-3:] == 'csv':
        df_nodes = pd.read_csv(df_nodes_file, header=None, sep=None)
    else:
        df_nodes = tt.find_node_coordiantes(df_nodes_file)

    df_nodes = df_nodes - df_nodes.mean(axis=0)
    df_nodes.index = np.arange(1, len(df_nodes)+1)
    df_nodes_values = df_nodes.values

    df_times_values = None
    df_speeds_values = None
    if isinstance(df_speeds_file, str) and len(df_speeds_file) == 2:
        df_speeds_file = f"demo/speeds-{df_speeds_file}.csv"
    if df_times_file is not None:
        df_times = pd.read_csv(df_times_file, header=None, sep=None)
        if df_times.shape[0] != df_times.shape[1]:
            df_times = tt.roll_table(df_times)
        df_times_values = df_times.values
    elif df_speeds_file is not None:
        df_speeds = pd.read_csv(df_speeds_file, header=None, sep=None)
        if df_speeds.shape[0] != df_speeds.shape[1]:
            df_speeds = tt.roll_table(df_speeds)
        df_speeds_values = df_speeds.values
    return df_nodes_values, df_times_values, df_speeds_values, df_nodes

@st.cache_data()
def do_tomo(method,df_nodes_values, df_times_values, df_speeds_values, 
               nx=None, ny=None, symmetric=False, fix_anisotropy=False):
    """ 
    Perform the tomography using the specified method.
    
            
    """
    logging.info(f"do_tomo entered for {method} method")
    kwds = {}
    if df_speeds_values is None or len (df_speeds_values) == 0:
        pass
    else:
        kwds['speeds'] = defix(df_speeds_values)
    if df_times_values is None or len (df_times_values) == 0:
        pass
    else:
        kwds['times'] = defix(df_times_values)
    if df_nodes_values is None or len (df_nodes_values) == 0:
        pass
    else:
        kwds['nodes'] = defix(df_nodes_values)
    t = tt.Tomo(symmetric=symmetric,
                   nx=nx, ny=nx, **kwds)
    if fix_anisotropy:
        t.fix_velocity_EBSI()
        logging.debug("Anisotropy fixed")
    else:
        t.fix_velocity_EBSI(lambda x,y:x)
        logging.debug("Anisotropy NOT fixed")
    if method == "EBSI":
        t.do_EBSI_method()
    elif method == "ERSM":
        t.do_ERSM_method()
    return t

def plot_angular_dependence(t, df_nodes_values, df_times_values, df_speeds_values, nx, anisotropy_handling, subplots=True, plotly=False):
    """
    Plot the angular dependence of the velocity.
    """
    logging.debug("Plotting angular dependence, anisotropy_handling = %s, subplots = %s, plotly = %s" % (anisotropy_handling, subplots, plotly)) 
    t['angle'] = do_tomo('angle', df_nodes_values, fix(df_times_values),
                            fix(df_speeds_values),
                            symmetric=False,
                            nx=21, ny=21, fix_anisotropy=anisotropy_handling)
    if plotly:
        return t['angle'].plot_angular_dependence(subplots=subplots, plotly=plotly)
    if subplots:
        figsize = (10, 18)
    else:
        figsize = None
    t['angle'].plot_angular_dependence(subplots=subplots, figsize=figsize, plotly=plotly)
    fig = plt.gcf()
    return fig

def handle_selection(key, all_selected, all_items, toggle_label):
    selected = st.session_state.get(f"selected_items_{key}", all_items)

        st.session_state[f"selected_items_{key}"] = [] if all_selected else all_items.copy()

        st.session_state[f"selected_items_{key}"] = [x for x in selected if x != toggle_label]

def make_pills(all_items, key="A", title="Select drilling positions for picture:"):
    items = st.session_state.get(f"selected_items_{key}", all_items)
    all_selected = set(items) == set(all_items)
    toggle_label = "Deselect All" if all_selected and all_items else "Select All"
    items_with_toggle = all_items + [toggle_label] if all_items else []
    st.session_state[f"selected_items_{key}"] = items
    st.pills(
        title,
        options=items_with_toggle,
        selection_mode="multi",
        key=f"selected_items_{key}",
        on_change=lambda: handle_selection(key, all_selected, all_items, toggle_label),
    )
