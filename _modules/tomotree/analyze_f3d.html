def get_xml_root(zip_path, inner_xml_filename):
    """Extract XML root from the F3D file."""
    with zipfile.ZipFile(zip_path, 'r') as archive:
        with archive.open(inner_xml_filename) as xml_file:
            tree = ET.parse(xml_file)
            return tree.getroot()

def extract_time_data(root):
    """Extract time of flight data from the XML root."""
    time_rows = root.findall(".//TimeRows//a")
    data = []
    for a in time_rows:
        row = [int(v.attrib['v']) for v in a.findall("./v")]
        data.append(row)
    return pd.DataFrame(data).reset_index(drop=True)

def extract_distance_data(root):
    """Extract distance data from the XML root."""
    sensor_data = []
    for sensor_node in root.iter("SensorPos_RawData"):
        if sensor_node.attrib.get("t") == "v":
            for v_elem in sensor_node.findall("v"):
                if v_elem.attrib.get("t") == "d":
                    val = v_elem.attrib.get("v")
                    if val is not None:
                        sensor_data.append(float(val))
    sensor_data = sensor_data[:-2]  # Remove last two elements
    index = [f"{i}-1" for i in range(2, 13)] + [f"{i}-7" for i in range(2, 13) if i != 7] 
    df = pd.DataFrame(sensor_data, index=index, columns=["distance"])
    return df

def df_nodes(df_distances, save=False, filename="nodes_coordinates.csv"):
    """
    Convert distance data to node coordinates.
    Also test if save is True, then save the coordinates to a CSV file.
    Ask if overwrite if the file already exists.
    """
    df = tomo.distances_to_coordiantes(df_distances)
    df = df - df.mean(axis=0)  # Center the coordinates around the mean
    if save:
        save_after_prompt(df, filename)
    return df

def distance_matrix(df_nodes):
    """Create a distance matrix from node coordinates."""
    t = tomo.Tomo(nodes=df_nodes, symmetric=True)
    d = np.zeros([len(df_nodes)] * 2)
    for ray in t.rays:
        d[ray.start.number - 1, ray.end.number - 1] = ray.length
        d[ray.end.number - 1, ray.start.number - 1] = ray.length
    distances_df = pd.DataFrame(d)
    return distances_df
