def transform_image(overlay, ax, A_img, B_img, A_target, B_target, alpha=1, zorder=500, plot_image=True):
    """
    Transformuje overlay obrázek (numpy pole) na `ax`, přičemž body A_img, B_img 
    v obrázku se zarovnají na A_target, B_target ve světových souřadnicích.

    Parametry:
    -----------
    overlay : ndarray
        PNG obrázek načtený jako numpy pole (např. pomocí `mpimg.imread`).
    ax : matplotlib.axes.Axes
        Osa, na kterou se má obrázek vykreslit.
    A_img, B_img : array-like
        Dva body v obrázku (v pixelech) – např. [x, y].
    A_target, B_target : array-like
        Cílové body ve světových souřadnicích.
    alpha : float
        Průhlednost obrázku.
    zorder : int
        Z-index pro překreslení.
    plot_image : bool
        Pokud je True, obrázek se vykreslí na osu `ax`.

    Návratová hodnota:
    -------------------
    trans : Affine2D
        Použitá transformace (v případě, že ji chceš znovu použít).
    """
    A_img = np.asarray(A_img)
    B_img = np.asarray(B_img)
    A_target = np.asarray(A_target)
    B_target = np.asarray(B_target)

    # Směrové vektory
    v_img = B_img - A_img
    v_target = B_target - A_target

    # Úhel natočení
    angle_img = np.arctan2(v_img[1], v_img[0])
    angle_target = np.arctan2(v_target[1], v_target[0])
    rotation_angle = angle_target - angle_img

    # Měřítko
    scale = np.linalg.norm(v_target) / np.linalg.norm(v_img)

    logging.debug(f"A_image is {A_img}")
    logging.debug(f"rotation_angle is {rotation_angle}")
    logging.debug(f"scale is {scale}")
    logging.debug(f"A_target is {A_target}")
    # Transformace
    trans = (
        Affine2D()
        .translate(-A_img[0], -A_img[1])      # 1. posun A_img na (0, 0)
        .rotate(rotation_angle)              # 2. otočení
        .scale(-scale, scale)                        # 3. změna měřítka
        .translate(*A_target)                # 4. posun na cílové místo
    )

    if plot_image:
        # Vykreslení overlay obrázku
        ax.imshow(overlay, transform=trans + ax.transData, alpha=alpha, zorder=zorder)

    return trans
