def get_directory(dirname="streamlit_data"):
    """
    The function `get_directory` allows users to select a data directory either from
    predefined directories or by uploading a ZIP file containing their own directory
    structure. It uses Streamlit for the user interface and caches the result for
    efficiency.
    """
    if "selected" not in st.session_state:
        st.session_state.selected = None
    if "rerun" not in st.session_state:
        st.session_state.rerun = False

    predefined_dirs = [d for d in os.listdir(dirname) if os.path.isdir(os.path.join(dirname, d))]
    predefined_dirs.sort()  # Sort directories for better UX

    if st.session_state.selected is None:

        st.title('TomoTree Web Interface')
        st.info("A python package for acoustic tomography of trees by Robert Ma≈ô√≠k and Valentino Cristini. *Supported by the Ministry of Education, Youth and Sports of the Czech Republic, project ERC CZ no. LL1909 ‚ÄúTree Dynamics: Understanding of Mechanical Response to Loading‚Äù*")

        st.markdown("## Data Selection or Upload")

        with st.sidebar:
            st.warning("Choose a data directory or upload a ZIP file containing your data.")
        st.markdown("### üìÅ Preloaded data")
        with st.container(border=True):
            col123 = st.columns(3)
            for i,d in enumerate(predefined_dirs):
                with col123[i % 3]:
                    with st.container(border=True):
                        # inssert the image {d}/thumbnail.png if it exists
                        thumbnail_path = os.path.join(dirname, d, "thumbnail.png")
                        if not os.path.exists(thumbnail_path):
                            thumbnail_path = "_static/placeholder.png"
                        st.image(thumbnail_path, caption=f"", use_container_width=True)
                        st.button(f"{d}".replace("_"," "), on_click=select_button, args=(f"{dirname}/{d}",), 
                              use_container_width=True, type="primary")
        st.markdown("### üì¶ Upload ZIP or f3d file")
        handle_zip_upload(dirname=dirname)
    else:
        with st.sidebar:
            if st.button("Reset and select another", use_container_width=True):
                st.session_state['speed_min'] = None
                st.session_state['speed_max'] = None
                st.session_state.selected = None
                st.session_state.rerun = True

    if st.session_state.rerun:
        st.session_state.rerun = False
        st.rerun()

    return st.session_state.selected

def handle_zip_upload(dirname=None):
    """
    Handles the upload and processing of a ZIP or .f3d file using Streamlit's file uploader.

    If a .f3d file is uploaded, a warning is shown, the file is saved to a temp folder and its content is printed.
    If a ZIP file is uploaded, it is extracted and passed to the `select_button` function.
    If no file is uploaded or an error occurs, nothing is returned.

    Parameters
    ----------
    dirname : str, optional
        The directory name where the uploaded files will be processed. Defaults to None.

    Returns
    -------
    str or None
        If a ZIP file is successfully uploaded and processed, returns the path to the extracted directory.
        If a .f3d file is uploaded, processes the file and returns None.
        If no file is uploaded or an error occurs, returns None.

    Notes
    -----
    Requires the `streamlit`, `tempfile`, `os`, `zipfile`, `numpy`, and `pandas` libraries.
    Assumes the presence of a `f3d` module with methods for processing .f3d files.
    Assumes the presence of a `select_button` function for further processing of the extracted directory.
    """
    uploaded_file = st.file_uploader("Upload your ZIP file", type=["zip","f3d"])
    if uploaded_file is not None:
        tmpdir = tempfile.mkdtemp()
        if uploaded_file.name.endswith('.f3d'):
            st.warning("The file is a .f3d file, which is not supported yet. Please upload a ZIP file.")
            # save the f3d file to the tmpdir
            f3d_path = os.path.join(tmpdir, uploaded_file.name)
            with open(f3d_path, "wb") as f:
                f.write(uploaded_file.getbuffer())
            root = f3d.get_root(f3d_path)
            df_distance = f3d.extract_distance_data(root)  # extract distances measured
            df_nodes_coords = f3d.df_nodes(df_distance) # find the coordinates of the nodes
            distances_df = f3d.distance_matrix(df_nodes_coords) # create a distance matrix
            df_tof = f3d.extract_time_data(root) # extract time of flight data
            splitdf = [df_tof.loc[df_tof.loc[:,i]==0,:].mean() for i in df_tof.columns] 
            df_tof_matrix = pd.concat(splitdf, axis=1)
            print("Nodes Coordinates:")
            print(df_nodes_coords)
            print("Time of Flight Matrix:")
            print(df_tof_matrix)
            # change diagonal to nan
            np.fill_diagonal(df_tof_matrix.values, np.nan)
            print(df_tof_matrix)
            print(df_tof_matrix-50)
            print("Distance Matrix:")
            print(distances_df)
            return None
        try:
            with zipfile.ZipFile(uploaded_file, "r") as zip_ref:
                zip_ref.extractall(tmpdir)

            subdirs = [os.path.join(tmpdir, name) for name in os.listdir(tmpdir)]
            dirs = [d for d in subdirs if os.path.isdir(d)]
            if len(dirs) == 1:
                target_dir = dirs[0]
            else:
                target_dir = tmpdir  # fallback

            st.success("Data from ZIP extracted.")
            select_button(target_dir)
            return target_dir

        except Exception as e:
            st.error(f"Error when processing ZIP file: {e}")
            return None
    return None
