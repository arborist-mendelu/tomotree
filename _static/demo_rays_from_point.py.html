x = np.linspace(-1.2,1.2,100)
y = np.linspace(-1.2,1.2,100)
xx,yy = np.meshgrid(x,y)
mynode = t2.nodes[1]
for ray in mynode.rays_from:
    color = tomo.speed2color(ray.speed, max=np.nanmax(data_velocity), min=np.nanmin(data_velocity))
    plt.plot(*list(zip(ray.start.coordinates, ray.end.coordinates)),
             color=color)
    # Vectorized version. We test all points in one function call. This is
    # much faster than nested loops.
    points = np.array([xx.reshape(-1),yy.reshape(-1)]).T
    points = points[tomo.is_point_inside_ellipse(points,
                    ray.start.coordinates, ray.end.coordinates,
                    eccentricity=tomo._e_value_EBSI(ray.start.number, ray.end.number, 12))]
    plt.plot(*points.T,".", color=color, alpha=0.15)
    plt.text(*ray.end.coordinates.T,round(ray.angle_from_radial*180/np.pi))
plt.gca().set_aspect(1)
