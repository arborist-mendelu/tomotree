from scipy.optimize import minimize

def find_time(ray,k=1,q=0):
    return sum([s.length / (k * s.speed + q) for s in ray.segments])

def sum_res(k=1, q=0):

def velocity_transformation(x):
    return sum_res(k=x[0],q=x[1])

x0 = [1,0]
opt = minimize(
    velocity_transformation, 
    x0=x0, 
    tol=velocity_transformation(x0)/10000
    )

g = t.grid
velocities_ERSM = [i.velocity for i in g.all_cells]

k, q = opt.x

for c in g.all_cells:
    c.velocity = c.velocity * k + q

t.plot(rays=False)
plt.suptitle("Rescaled ERSM method")
velocities_new = [i.velocity for i in g.all_cells]
