fig, ax = plt.subplots(1,2, figsize=(12,6))

g = tomo.Grid(nx=15,ny=15, xlim=(-6,6), ylim=(-3,11))
g.get_cell(1,1).velocity = 2000
g.get_cell(-.5, 3).velocity = 200
g.get_cell(-2,6).velocity = 1000

n1 = tomo.Node(np.array([-4,-2]), 1)
n2 = tomo.Node(np.array([2,8]), 2)
n3 = tomo.Node(np.array([-5,10]), 3)
n4 = tomo.Node(np.array([4.3,2]), 4)
r1 = tomo.Ray(n1,n2,1700)
r2 = tomo.Ray(n3,n4,600)
g._save_rays_info_to_cells([r1,r2])
g.set_cell_velocity_from_ray_means()
g.plot(ax=ax[0], shading='flat')

cell = g.get_cell(-2, 9)
ax[0].plot(*cell.boundary_points.T)

g._clear_velocities_list()       
g.get_cell(-2, 0).velocities_list = [10,0,1,2]

cells_bp = np.array(list(map(lambda x:x.boundary_points, g.all_cells)))
c = np.array(g.all_cells)
for ray in [r1,r2]:
    c_subset = c[tomo.is_point_inside_ellipse(cells_bp, ray.start.coordinates, ray.end.coordinates, 0.95).any(axis=1)]
    for i in c_subset:
        i.velocities_list += [ray.speed]

for cell in g.all_cells:
    if len(cell.velocities_list)>0:
        cell.velocity = np.nanmean(cell.velocities_list)
    else:
        cell.velocity = np.nan
    
g.plot(ax=ax[1], shading='flat')
for a in ax:
    for r in [r1,r2]:
        a.plot(*(np.array([r.start.coordinates, r.end.coordinates]).T), lw=3, color='k')    
ax[1].vlines(g.x,min(g.y),max(g.y))
ax[1].hlines(g.y,min(g.x),max(g.x))

ax[0].set(title="Rays in grid")
ax[1].set(title="Cells influenced by the rays")
ax[0].legend(["Cell boundary"])
