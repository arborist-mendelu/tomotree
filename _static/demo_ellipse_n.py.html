e = 0.95
start = time.time()

ray = tomo.Ray(
    start=tomo.Node(np.array([0,-1]),1),
    end=tomo.Node(np.array([-3,1]),2) 
    )
segment = list(ray.segments)[0]
segment.split(5)
fig,ax = plt.subplots()
plt.plot(*list(zip(ray.start.coordinates,ray.end.coordinates)), lw=4)

x = np.linspace(-3.1,  .2,200)
y = np.linspace(-1.2, 1.2,200)
xx,yy = np.meshgrid(x,y)
points = np.array([xx.reshape(-1),yy.reshape(-1)]).T

idx_r,d_r = tomo.is_point_inside_ellipse(points, 
    ray.start.coordinates, ray.end.coordinates, 
    eccentricity=e, distance_from_vertex=True)
p = points[idx_r]
ax.plot(*p.T, alpha=0.2)
ax.grid()

segments = list(ray.segments)
segments.sort(key = lambda x:x.start)
for i,s in enumerate(segments):    
    idx = tomo.is_point_inside_ellipse(points, s.start_coordinates, 
                                       s.end_coordinates, eccentricity=e)
    p = points[idx]
    ax.plot(*p.T,'o', color=f"C{i}", alpha=0.8)

    idx2 = ( d_r>s.ray.length*(s.start) ) & ( d_r<s.ray.length*(s.end) )
    ax.plot(*points[idx_r & idx2].T,'.', color=f"C{i}", alpha=0.1)
    ax.text(*(0.5*(s.start_coordinates + s.end_coordinates)),f"{i}: {s.start:.2f}-{s.end:.2f}")

ax.set_aspect(1)
plt.title("Ray divided into 5 segments")

print (f"Finished in {time.time()-start:.4f} sec.")
