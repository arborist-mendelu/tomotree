from matplotlib.colors import LinearSegmentedColormap
colors = ['aqua', 'red', 'yellow', 'lawngreen', 'darkgreen']
cmap = LinearSegmentedColormap.from_list('custom_colormap', colors)


def plot(self, ax=None, speed_min=0, speed_max=2000, shading='flat', use_contourf: bool=False, **kwds):
    """Plot a heatmap of the grid velocities using `pcolormesh` or `contourf`.

    Return:
        The axes object containing the heatmap plot.
    """
    if ax is None:
        fig, ax = plt.subplots()
    data = self.cells.iloc[:-1, :-1].map(lambda x: x.velocity).values
    if use_contourf:
        shading = ''
    if shading == 'flat':
        x = self.x
        y = self.y
    else:
        x = self.x[:-1] + 0.5 * self.dx
        y = self.y[:-1] + 0.5 * self.dy
    if use_contourf:
        t = ax.contourf(x,y,data.T, cmap=cmap, **kwds)
        ax.contour(t,colors='k')
        # ax.clabel(t, colors='k')
    else:
        t = ax.pcolormesh(x, y, gaussian_filter(data.T, sigma=1), shading=shading,
                      cmap=cmap, vmin=speed_min, vmax=speed_max)
    
    plt.colorbar(t)
    ax.set_aspect(1)
    return ax
