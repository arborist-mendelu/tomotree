import sys
import matplotlib.pyplot as plt
sys.path.append('../src')
import tomotree as tomo
import pandas as pd 
DIRECTORY = '../streamlit_data/Jin_05/'
nodes_file = DIRECTORY + 'nodes-05.csv'
speeds_file = DIRECTORY + 'speeds-05.csv'
df_nodes = pd.read_csv(nodes_file, header=None)
df_speeds = pd.read_csv(speeds_file, header=None)

data = {'nodes':df_nodes.values, 'speeds': df_speeds.values, 'symmetric':True}
plot_data = {'boundary':'spline',             
             'plot_colorbar':False
}
In[2]:
fig, axs = plt.subplots(2,3, figsize=(15,10))

ax = axs[0,0]
t = tomo.Tomo(**data, nx=10, ny=10)
t.plot(rays=True, grid=False, **plot_data, ax=ax)
ax.set(title="Acoustic rays visualization")

ax = axs[0,1]
t = tomo.Tomo(**data, nx=8, ny=8)
t.plot(rays=False, grid=True, ax=ax, **plot_data)
ax.set(title="Basic back projection method, coarse grid")

ax = axs[0,2]
t = tomo.Tomo(**data, nx=80, ny=80)
t.do_EBSI_method()
t.plot(rays=False, grid=True, **plot_data, 
            speed_max=1500, 
            speed_min=900, 
            ax=ax
            )
ax.set(title="EBSI method")


ax = axs[1,0]
t = tomo.Tomo(**data, nx=100, ny=100)
t.do_ERSM_method()
t.plot(rays=True, grid=False, speed_max=1500, speed_min=1000, 
       ax=ax, **plot_data)
ax.set(title='ERSM method, rays segmentation')


ax = axs[1,1]
t = tomo.Tomo(**data, nx=100, ny=100)
t.do_ERSM_method()
t.plot(rays=False, grid=True, speed_max=1400, speed_min=1000, 
       ax=ax, **plot_data)
ax.set(title='ERSM method')

ax = axs[1,2]
t = tomo.Tomo(**data, nx=100, ny=100)
t.do_ERSM_method()
t.grid.set_cell_status_from_threshold(1100, nan_as_zero=True)
t.plot(rays=False, grid=True, status=False, speed_max=1400, speed_min=1000,
       ax=ax, **plot_data)
t.plot(rays=False, grid=True, status=True, speed_max=1400, speed_min=1000, **plot_data, ax=ax)
ax.set(title='Binary mask for ERSM reconstruction')
