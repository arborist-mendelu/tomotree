data = pd.read_csv('tomoWood_data/1_posRecHealthy.txt', sep='\t', header=None)
data = data - data.mean()
data_nodes = data.copy()
plt.plot(*data_nodes.values.T,"o")

data = np.full((data_nodes.shape[0], data_nodes.shape[0]),1000)

t = tomo.Tomo(nodes=data_nodes.values, 
              speeds=data, symmetric=True)
_speeds = {"speed_min":400, "speed_max":1200}
t.do_SIRT_method(**_speeds)

for c in t.grid.all_cells:
    if c.x > 0.02 and c.x < 0.1 and c.y > -0.05 and c.y < 0.05:
        c.velocity = 500
        for s in c.segments:
            s.speed = 500
    if c.x > -0.1 and c.x < -0.05 and c.y > 0 and c.y < 0.05:
        c.velocity = 500
        for s in c.segments:
            s.speed = 500

idx = np.arange(1,len(data_nodes.values)+1)
idx
times = pd.DataFrame(columns=idx, index = idx)
for r in t.rays:
    start = r.start.number
    end = r.end.number
    time = sum([s.length/s.speed for s in r.segments])
    s1 = sum([s.length for s in r.segments])
    s2 = r.length
    if np.abs(s1-s2)>1e-12:
        print("Problem, lengths do not match.",start, end, time)
    times.at[start,end] = time
    times.at[end, start] = time
