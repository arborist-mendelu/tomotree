from scipy.optimize import minimize

def find_time(ray,x):
    return sum([s.length / sum(x* (s.speed**np.arange(4))) for s in ray.segments])

def sum_res(t, x=[0,1,0,0]):
    return sum([(find_time(i,x) - i.time)**2 for i in t.rays])

def velocity_transformation(t,x):
    return sum_res(t,x)
