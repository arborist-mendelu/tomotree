import sys
import matplotlib.pyplot as plt
sys.path.append('../src')
import tomotree as tomo
import pandas as pd 
DIRECTORY = '../streamlit_data/Jin_08/'
nodes_file = DIRECTORY + 'nodes-08.csv'
speeds_file = DIRECTORY + 'speeds-08.csv'
df_nodes = pd.read_csv(nodes_file, header=None)
df_speeds = pd.read_csv(speeds_file, header=None)

data = {'nodes':df_nodes.values, 'speeds': df_speeds.values, 'symmetric':True, 
        'nx':100, 'ny':100, 
        'boundary':'polygon'}
plot_data = {'boundary':'polygon',             
             'plot_colorbar':False, 
             'speed_max':1200, 
             'speed_min':600,
}
